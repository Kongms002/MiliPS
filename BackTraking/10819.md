```
#include <bits/stdc++.h>

using namespace std;

int N;
vector<int> arr;
vector<int> result;
int visited[8] = {
    0,
};

int answer = 0;
void dfs(int cnt) {
  if (cnt == N) {
    int sum = 0;
    for (int i = 0; i < N - 1; i++)
      sum += abs(result[i] - result[i + 1]);
    answer = max(sum, answer);
    return;
  }

  for (int i = 0; i < N; i++) {
    if (!visited[i]) {
      visited[i] = 1;
      result.push_back(arr[i]);
      dfs(cnt + 1);
      result.pop_back();
      visited[i] = 0;
    }
  }
}

int main() {
  cin >> N;
  for (int i = 0; i < N; i++) {
    int x;
    cin >> x;
    arr.push_back(x);
  }
  dfs(0);
  cout << answer;
  return 0;
}
```
