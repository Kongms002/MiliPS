#include <bits/stdc++.h>

using namespace std;

int main(){
  vector<int> dp(1000001, 0);

  int N; cin >> N;

  vector<int> next(N+1, 0);
  dp[1] = 0; 

  for(int i=2;i<=N;i++){

    int t = int(10e6);
    if(i%3==0 && dp[i/3] < t) {
      t = dp[i/3];
      next[i] = i/3;
    }
    if(i%2==0 && dp[i/2] < t) {
      t = dp[i/2];
      next[i] = i/2;
    }
    if(dp[i-1] < t){
      t = dp[i-1];
      next[i] = i-1;
    }
    dp[i] = t+1;
  }
  cout << dp[N] << "\n";
  while(next[N] >= 1){
    cout << N << " ";
    N = next[N];
  }
  cout << "1";
  return 0;
}
